\begin{frame}{Why this talk: can we drop the sparse side?}
  \textbf{Production retrieval today is \alert{hybrid}}: a dense embedding index
  (HNSW / IVF) \emph{plus} an inverted index (BM25). Two stores, two scoring
  paths, extra ops.

  \medskip
  \textbf{The tempting question}: could a pure-dense stack replace hybrid?
  Simpler, one index, one score.

  \medskip
  Two things have to hold:
  \begin{itemize}
    \item \textbf{Bottleneck 1 --- representation.}
          Can the embedding \emph{encode} what BM25 encodes?
          Well-studied: Tay et al.\ 2020, Fan et al.\ 2022 --- capacity grows with $d$.
    \item \textbf{Bottleneck 2 --- indexing.}
          Even if the embedding is capable, can an \emph{ANN index} still find it?
          \alert{This talk. Largely unstudied.}
  \end{itemize}

  \vspace{0.5em}
  \begin{block}{Headline (the index alone --- same embedding)}
    \textbf{Flat $\to$ IVF} (partition index the theory targets):\;
    forfeits \alert{$\sim 33\%$} of achievable Recall@10 on identifier queries,
    $<5\%$ on standard.\\[0.2em]
    \textbf{Flat $\to$ HNSW} (graph index, same population):
    \alert{$\sim 63\%$} on identifier queries, $<2\%$ on standard.
  \end{block}
\end{frame}

%% =====================================================================
\begin{frame}{The population that pays the bill: ``spear-fishing'' queries}
  \begin{block}{Working definition}
    Queries whose relevance hinges on a single, low-frequency, high-IDF token:
    identifiers (SKUs, part numbers), rare terms, precise named entities.
  \end{block}

  \begin{itemize}
    \item Small fraction of traffic, but operationally critical:
          failed spear-fishing $\Rightarrow$ empty result pages, abandoned carts,
          broken RAG lookups.
    \item BM25 handles them trivially (exact token match).
    \item If pure dense is going to replace hybrid, it has to hold up \emph{here}.
    \item Our three query subsets: \textbf{standard} (control),
          \textbf{rare} ($\le 5$ docs), \textbf{ID} (extreme case: doc identifier
          as query).
  \end{itemize}
\end{frame}

%% =====================================================================
\begin{frame}{The idealized case: BM25 \emph{is} embedding search}
  Reframe BM25 as a dense inner product in a huge, discrete embedding space
  $\mathbb{R}^V$ --- \emph{one dimension per token}
  ($V$ = vocabulary size; for our char-trigrams, order $10^6$--$10^7$).

  \medskip
  Let $M$ = number of docs in the corpus and $n_t$ = number of docs containing
  token $t$. Each doc $d$ becomes a sparse, IDF-weighted vector:
  \[
    f(d)_t \;=\; \alpha_t \;=\; \log\!\big(M / n_t\big)
    \quad\text{if } t \in d, \text{ else } 0.
  \]
  A query is a \emph{binary indicator}: $q_t = 1$ if $t \in q$, else $0$.

  \medskip
  For a query $q = \{t^*\}$ with rare token $t^*$:
  \begin{itemize}
    \item Relevant doc $d_{\text{rel}}$ (contains $t^*$):
          $\;\langle q, f(d_{\text{rel}})\rangle = \alpha_{t^*}$
          --- \alert{big}, because $n_{t^*}$ is small.
    \item Irrelevant doc: $\langle q, f(d)\rangle = 0$.
  \end{itemize}
  \alert{Perfect separation on spear-fishing.} That is why BM25 works ---
  it is nearest-neighbor search in $\mathbb{R}^V$.

  \medskip
  \textbf{Catch}: $V$ is enormous. BM25 handles it with an inverted list ---
  ANN cannot. To use ANN we must compress $\mathbb{R}^V \to \mathbb{R}^k$
  with $k \ll V$. \emph{What compression costs is the rest of the talk.}
\end{frame}

%% =====================================================================
\begin{frame}{Compressing $\mathbb{R}^V \to \mathbb{R}^k$: Johnson--Lindenstrauss}
  Random linear map $A: \mathbb{R}^V \to \mathbb{R}^k$, target dimension
  $k \in \{256, 512, 1024\}$ (our embedding size --- distinct from the
  document symbol $d$).

  \begin{block}{JL theorem (informal)}
    Given $M$ vectors in $\mathbb{R}^V$, a random projection $A$ preserves
    all pairwise inner products \emph{up to a multiplicative error} $\epsilon$
    with high probability, provided
    \[
      k \;\gtrsim\; \frac{\log M}{\epsilon^2},
      \qquad\text{equivalently}\qquad
      \epsilon \;\approx\; \sqrt{\tfrac{\log V}{k}}.
    \]
  \end{block}

  \textbf{What survives, what dies}:
  \begin{itemize}
    \item Inner products with magnitude \alert{$\gg \epsilon$} are preserved.
          $\Rightarrow$ Spear-fishing signal $\alpha_{t^*}$ from the previous
          slide easily clears $\epsilon$: \textbf{Flat search on the projected
          embedding is a clean oracle.}
    \item Magnitudes \alert{$\lesssim \epsilon$} are drowned in projection
          noise --- indistinguishable from random.
    \item Bigger $k$ shrinks $\epsilon$; nothing else does.
  \end{itemize}

  \medskip
  \small So after JL alone, spear-fishing still works. The failure emerges
  when we add the \emph{second} compression --- clustering --- on top. IVF.
\end{frame}

\begin{frame}{Quick primer: what is an IVF index?}
  \textbf{IVF} (Inverted File) --- the standard partition-based ANN index.
  \begin{enumerate}
    \item Cluster the corpus into $N$ cells (k-means). Each cell has a
          \emph{centroid} $c$.
    \item At query time: score the query against all $N$ centroids;
          keep the top $n_{\text{probe}}$.
    \item Only search documents inside those $n_{\text{probe}}$ cells.
  \end{enumerate}

  \begin{center}
    \begin{tikzpicture}[scale=0.9]
      \fill[good!15] (0.5,0.5) circle (0.9);
      \fill[good!15] (3.0,1.2) circle (0.9);
      \fill[accent!25] (5.5,0.6) circle (0.9);
      \fill[good!15] (2.0,2.7) circle (0.9);
      \foreach \x/\y in {0.4/0.4, 0.7/0.9, 0.3/0.7, 0.9/0.5,
                         2.8/1.0, 3.1/1.4, 3.3/1.1, 2.9/1.6,
                         5.3/0.5, 5.7/0.9, 5.6/0.4,
                         1.8/2.5, 2.2/2.9, 2.0/2.5}
        \fill[gray!70] (\x,\y) circle (0.04);
      \fill[good] (0.5,0.5) circle (0.09);
      \fill[good] (3.0,1.2) circle (0.09);
      \fill[accent] (5.5,0.6) circle (0.09);
      \fill[good] (2.0,2.7) circle (0.09);
      \fill[black] (5.4,0.9) circle (0.11);
      \node[right] at (5.55,0.9) {\scriptsize $q$};
      \draw[dashed,accent,thick] (5.4,0.9) -- (5.5,0.6);
      \node[below] at (0.5,0.4) {\scriptsize $c_1$};

      \node[below] at (5.5,0.5) {\scriptsize $c_3$};
      \node[below] at (2.0,2.6) {\scriptsize $c_4$};
    \end{tikzpicture}
  \end{center}

  \vspace{-0.4em}
  \small
  \textbf{Speedup}: $N / n_{\text{probe}}$ fewer distance computations.\quad
  \textbf{Failure mode}: if the relevant cell's centroid is not in the top
  $n_{\text{probe}}$ by query similarity, its documents are \emph{invisible}.
\end{frame}

%% =====================================================================
\begin{frame}{The mechanism: centroid dilution (intuition)}
  A centroid is the \emph{average} of the vectors in its cell.

  \medskip
  \begin{itemize}
    \item A \emph{common} token appears in many docs $\Rightarrow$ contributes
          strongly to the average $\Rightarrow$ shows up in the centroid.
    \item A \emph{rare} token appears in a handful of docs $\Rightarrow$ its
          contribution is averaged \emph{away} $\Rightarrow$ nearly invisible
          in the centroid.
  \end{itemize}

  \medskip
  Now consider a query dominated by that rare token:
  \begin{itemize}
    \item Query vector is nonzero on the rare-token direction.
    \item Every centroid --- \emph{including the one containing the relevant docs} ---
          has \emph{almost zero} weight on that direction (averaged away).
    \item Result: \alert{all centroids look equally (un)related to the query}.
          Cluster selection becomes essentially random.
  \end{itemize}

  \medskip
  \emph{The failure is not the embedding. It is the averaging step in the index.}
\end{frame}

%% =====================================================================
\begin{frame}{The bound: when does the centroid signal survive?}
  \footnotesize \emph{So far}: in $\mathbb{R}^V$ the relevant doc scores
  $\alpha_{t^*}$; JL preserves it. But \textbf{IVF selects clusters by the
  centroid}, and a centroid is an average.

  \small
  \textbf{Step 1 --- Centroid.} Partition into $N$ clusters, average size
  $|C| \approx M/N$:
  \(
    c_C = \tfrac{1}{|C|} \sum_{d \in C} f(d).
  \)

  \textbf{Step 2 --- Prevalence.} Let $n_{t^*}$ = \# docs \emph{in $C$}
  containing $t^*$. Only these contribute to the $t^*$-direction of $c_C$:
  \[
    c_C \big|_{t^*}
    \;\approx\;
    \underbrace{\tfrac{n_{t^*}}{|C|}}_{\displaystyle p\;=\;\text{prevalence of }t^*\text{ in }C}
    \;\cdot\; \alpha_{t^*}.
  \]
  Rare token $\Rightarrow$ $p$ tiny (often $1/|C|$, or $0$).

  \textbf{Step 3 --- Query.} $q$ = binary indicator of $t^*$ ($q_{t^*}=1$):
  \[
    \boxed{\;\langle q,\, c_C \rangle \;\approx\; \alpha_{t^*} \cdot p\;}
  \]

  \begin{block}{Survival condition}
    \footnotesize\centering
    Relevant centroid picked \emph{only when} $\;\alpha_{t^*}\cdot p \gg \epsilon\;$
    (JL floor). Rare: $\alpha$ big, $p$ tiny $\Rightarrow$ product
    $<\epsilon$ $\Rightarrow$ wrong cluster.
    Bigger $k$ helps $\epsilon$; nothing helps $p$.
  \end{block}
\end{frame}

%% =====================================================================
\begin{frame}{HNSW isn't a way out}
  \textbf{HNSW} is not partition-based --- it is a \emph{layered proximity graph}.
  Greedy descent from a hub node down to the query's local neighborhood.

  \medskip
  \textbf{So why does it fail on the same queries?}
  \begin{itemize}
    \item Graph edges are built from Euclidean distances during insertion.
    \item Dense regions of the embedding space get thick, well-connected
          subgraphs. Rare vectors land in sparse regions --- \emph{few incoming
          edges from any hub}.
    \item Greedy search from a hub cannot reach an orphan node without a long
          detour: it stops at a locally-good but globally-wrong region.
    \item Widening the beam (\texttt{efSearch}) helps --- but with diminishing
          returns.
  \end{itemize}

  \medskip
  \textbf{The pattern is the same}: rare-token vectors are structurally hard
  to \emph{reach}, whether the structure is a partition (IVF) or a graph (HNSW).
  Different mechanism; \alert{same victims}.
\end{frame}

\begin{frame}{Experimental setup (compact)}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \textbf{Datasets} (1{,}000 queries per subset)
    \begin{itemize}
      \item Home Depot (HD): 54{,}667 product titles
      \item MS MARCO Passage v1.1: 82{,}360 passages
    \end{itemize}

    \textbf{Query subsets}
    \begin{itemize}
      \item standard $\;\vert\;$ rare ($\le 5$ docs) $\;\vert\;$ ID
    \end{itemize}

    \column{0.45\textwidth}
    \textbf{Representation}: char-trigram + JL projection,
    $d\in\{256,512,1024\}$, unsupervised.

    \textbf{Indices (FAISS)}: Flat, IVF, HNSW ($M{=}16$, $\text{efC}{=}200$),
    BM25 baseline (Pyserini).
  \end{columns}

  \vspace{0.8em}
  \begin{block}{Why unsupervised trigrams?}
    A supervised encoder abstracts away the lexical detail that \emph{defines}
    spear-fishing queries. To isolate the \emph{index} bottleneck from the
    representation bottleneck we need a representation that provably preserves
    lexical distances (Johnson--Lindenstrauss).
  \end{block}
\end{frame}

\begin{frame}{Direct evidence: measure the centroid signal}
  \textbf{For each query}: compute $\langle q,\, c_{\text{rel}}\rangle$
  on the centroid of the query's \emph{relevant} cluster
  (IVF, $n_{\text{list}}{=}500$, $d{=}1024$, HD).

  \begin{columns}[T]
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/centroid_signal_by_subset.png}

    \column{0.45\textwidth}
    \small
    \setlength{\tabcolsep}{3pt}
    \begin{tabular}{lrrr}
      \toprule
      Subset & signal & rank & hit@250 \\
      \midrule
      standard & 0.279 & 2   & 97\% \\
      rare     & 0.066 & 69  & 81\% \\
      \alert{id} & \alert{0.007} & \alert{216} & \alert{59\%} \\
      \bottomrule
    \end{tabular}

    \medskip
    \alert{$\sim 40\times$ collapse} of the centroid signal from standard to ID
    --- the correct cluster is buried at rank $\sim 200$.
  \end{columns}

  \medskip
  \small The bound $\langle c,q\rangle \approx \alpha_{t^*}\cdot p$ is not just
  suggestive --- it is what we \emph{measure}.

  \note{%
  \textbf{Framing.} This is where we stop arguing from theory and start
  measuring the exact quantity the bound predicts.

  \textbf{What we did (per query).} Train IVF on Home Depot ($n_{\text{list}}=500$,
  $d=1024$). For every query, look up which cluster actually contains its
  relevant doc(s) --- call that centroid $c_{\text{rel}}$. Then compute
  $\langle q, c_{\text{rel}}\rangle$, and also \emph{where} $c_{\text{rel}}$ ranks
  among all 500 centroids scored against $q$, and whether it makes the top 250
  (``hit@250'' --- the halfway mark).

  \textbf{Histogram.} Standard queries: signal $\sim 0.28$, clearly separated
  from zero. Rare queries collapse toward zero ($\sim 0.07$). ID queries pile
  up at $\sim 0.007$ --- \textbf{40$\times$ smaller than standard}.

  \textbf{Table.} Standard: correct centroid at rank 2 on average, hit@250
  = 97\% --- IVF with $n_{\text{probe}}=5$ is basically always correct. Rare:
  rank 69, so you'd need $n_{\text{probe}}\approx 70$ (huge). \textbf{ID:
  rank 216 out of 500 --- worse than median.} hit@250 = 59\% means
  \textbf{41\% of ID queries have their relevant cluster in the bottom half}
  of the centroid ranking. Those are structurally unrecoverable short of
  scanning all clusters --- which defeats IVF's purpose.

  \textbf{Tie to the bound.} $\langle c,q\rangle \approx \alpha_{t^*}\cdot p$.
  Rare tokens: huge $\alpha$ (IDF), tiny $p$ (prevalence). Product falls
  under the JL floor $\epsilon$ --- exactly the collapse we measure.

  \textbf{Anticipated question.} ``Just use finer clusters?'' It helps
  (rare/ID gain 2--3$\times$ more relatively than standard going
  $n_{\text{list}}$ 200$\to$1000) but you're fighting $p$: shrinking clusters
  means more centroids to score, i.e.\ slower search. That's the
  tradeoff Slide 11 quantifies.

  \textbf{Time cue.} Spend a full minute on the two numbers ``rank 216''
  and ``41\% unrecoverable'' --- this is the empirical hinge of the paper.
  }
\end{frame}

\begin{frame}{The production-visible symptom}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/D_zero_rate.png}

    \column{0.45\textwidth}
    \textbf{Fraction of queries with zero relevant docs in top-10:}
    \begin{itemize}
      \item HD-id: Flat \textbf{13\%} $\to$ HNSW \alert{\textbf{71\%}}
      \item MM-id: Flat \textbf{78\%} $\to$ HNSW \alert{\textbf{92\%}}
      \item Standard: within $\pm 2$ pp on both datasets.
    \end{itemize}

    \medskip
    \textbf{Cost to match Flat recall} (efSearch, HD):
    \begin{itemize}
      \item standard: \textbf{2.7$\times$ faster} than Flat
      \item id: \alert{\textbf{40$\times$ slower}} than Flat
    \end{itemize}
    ANN's speedup evaporates on the population that needs it most.
  \end{columns}

  \note{%
  \textbf{Framing.} We've shown the mechanism (theory) and the direct
  measurement (centroid signal). This slide is what the \emph{user} actually
  sees.

  \textbf{Left: empty-result rate ($D_0$).}
  $D_0$ = fraction of queries whose top-10 contains \emph{zero} relevant docs
  --- an empty search page from the user's point of view.
  \begin{itemize}
    \item HD-ID: Flat 13\% $\to$ HNSW \textbf{71\%}. Seven of ten ID queries
          return nothing relevant on HNSW.
    \item MM-ID: 78\% $\to$ 92\%. Harder dataset, weaker absolute signal.
    \item Standard queries: within $\pm 2$ pp between Flat and HNSW on
          \emph{both} datasets --- mainstream traffic sees nothing.
  \end{itemize}
  So the failure is \textbf{concentrated on spear-fishing traffic and
  invisible in aggregate metrics}. Aggregate recall looks fine, product
  looks broken for the SKU-searching user.

  \textbf{Right: cost-to-close ratio.} Same phenomenon, dual angle.
  Instead of ``at default efSearch what recall do we get?'' ask ``how
  much efSearch to \emph{match} Flat's recall?''
  \begin{itemize}
    \item Standard queries: efSearch=100 hits 95\% of Flat recall at
          0.39 ms/q --- 2.7$\times$ \emph{faster} than Flat. Canonical ANN
          speedup, working as advertised.
    \item ID queries: efSearch=8000 needed, 41.9 ms/q --- \textbf{40$\times$
          slower than Flat} on the same corpus. On MS MARCO the ratio
          reaches $\sim 80\times$.
  \end{itemize}
  HNSW's value proposition (``$\approx$Flat recall, way faster'')
  \textbf{inverts} on IDs: same recall, way slower. You're paying more
  for the same accuracy on exactly the queries you needed acceleration for.

  \textbf{Corpus generality.} efSearch cost scales roughly linearly in
  corpus size, so this ratio ports to any $N$ --- not a quirk of 54k or 82k
  documents.

  \textbf{Landing line for the audience.} If you deploy pure-dense with
  the standard HNSW recipe and tune efSearch for good aggregate recall
  at low latency, you are \emph{implicitly deciding} to let 60--90\% of
  identifier traffic return empty. That decision is invisible in your
  aggregate metrics and highly visible to the user searching a specific
  SKU. This is why hybrid isn't belt-and-suspenders --- the sparse side
  is load-bearing on a population dense cannot serve at practical latency.
  }
\end{frame}

\begin{frame}{``But learned encoders would fix this''}
  Anticipated question: \emph{do supervised dense encoders escape the issue?}

  \medskip
  Re-ran with \texttt{all-MiniLM-L6-v2} (Sentence-BERT, 384-dim):

  \begin{center}
    \small
    \setlength{\tabcolsep}{6pt}
    \begin{tabular}{llrr}
      \toprule
      Dataset & Subset & Trigram Flat R@10 & \textbf{MiniLM Flat R@10} \\
      \midrule
      HD       & standard & 0.327 & 0.298 \\
      HD       & rare     & 0.751 & 0.428 \\
      HD       & {\color{accent}id} & {\color{good}0.874} & {\color{accent}\textbf{0.010}} \\
      MS MARCO & standard & 0.471 & {\color{good}\textbf{0.894}} \\
      MS MARCO & {\color{accent}id} & {\color{good}0.174} & {\color{accent}\textbf{0.001}} \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{itemize}
    \item MiniLM \emph{cannot distinguish identifiers} --- ``12345'' $\approx$ ``67890''.
    \item Representation collapses \emph{before} the index has a chance to bias.
    \item Which is precisely why hybrid retrieval exists.
  \end{itemize}
\end{frame}

%% =====================================================================
\begin{frame}{Takeaways}
  \begin{enumerate}
    \item Production dense retrieval is hybrid for a reason. If you want to
          drop the sparse side, you have to clear \emph{two} bottlenecks, not one.
    \item Bottleneck 1 (representation) is well-studied. Bottleneck 2
          (indexing) is what this paper isolates and quantifies.
    \item Centroid dilution + JL noise floor give a closed-form bound; the
          $\sim 40\times$ centroid-signal collapse from standard to ID is
          exactly what the bound predicts.
    \item HNSW is not a workaround --- the mechanism differs, the victims
          don't.
    \item Practical implication: query-type-aware routing, index-aware
          training, or --- most simply --- keep the sparse side.
  \end{enumerate}
